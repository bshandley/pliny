```typescript
// Subtask counts - how many subtasks each card has and how many are "done"
// "Done" = in the rightmost column of the board
const subtaskCountsQuery = `
  SELECT
    c.parent_card_id,
    COUNT(*)::int as total,
    COUNT(*) FILTER (WHERE c.column_id = (
      SELECT id FROM columns
      WHERE board_id = $1
      ORDER BY position DESC
      LIMIT 1
    ))::int as done
  FROM cards c
  INNER JOIN columns col ON c.column_id = col.id
  WHERE col.board_id = $1 AND c.parent_card_id IS NOT NULL
  GROUP BY c.parent_card_id
`;
const subtaskCountsResult = await pool.query(subtaskCountsQuery, [boardId]);
const subtaskCountsMap = new Map<string, { total: number; done: number }>();
for (const row of subtaskCountsResult.rows) {
  subtaskCountsMap.set(row.parent_card_id, { total: row.total, done: row.done });
}
```

**Step 2: Add POST /cards/:id/subtasks route**

```typescript
// Create a subtask
router.post('/:id/subtasks', authenticate, requireBoardRole('EDITOR'), async (req, res) => {
  const { id: parentId } = req.params;
  const { title, column_id } = req.body;

  // Validate parent exists and is not a subtask
  const parentResult = await pool.query(`
    SELECT c.id, c.parent_card_id, col.board_id
    FROM cards c
    JOIN columns col ON c.column_id = col.id
    WHERE c.id = $1
  `, [parentId]);

  if (parentResult.rows.length === 0) {
    return res.status(404).json({ error: 'Parent card not found' });
  }
  if (parentResult.rows[0].parent_card_id) {
    return res.status(400).json({ error: 'Cannot add subtask to a subtask' });
  }

  // Validate column is in same board
  const colResult = await pool.query('SELECT board_id FROM columns WHERE id = $1', [column_id]);
  if (colResult.rows.length === 0 || colResult.rows[0].board_id !== parentResult.rows[0].board_id) {
    return res.status(400).json({ error: 'Column must be in same board as parent' });
  }

  // Get max position
  const posResult = await pool.query(
    'SELECT COALESCE(MAX(position), -1) + 1 as next_pos FROM cards WHERE column_id = $1',
    [column_id]
  );

  const result = await pool.query(`
    INSERT INTO cards (title, column_id, position, parent_card_id)
    VALUES ($1, $2, $3, $4)
    RETURNING *
  `, [title, column_id, posResult.rows[0].next_pos, parentId]);

  // Log activity
  await logActivity(pool, parentResult.rows[0].board_id, result.rows[0].id, req.user!.id, 'created', 'card');

  res.status(201).json(result.rows[0]);
});
```

```typescript
const handleToggleSubtaskDone = async (subtask: SubtaskSummary) => {
  if (!columns || columns.length === 0) return;

  // Find rightmost and leftmost columns
  const sortedCols = [...columns].sort((a, b) => a.position - b.position);
  const lastCol = sortedCols[sortedCols.length - 1];
  const firstCol = sortedCols[0];

  // If in last column, move to first; otherwise move to last
  const targetColId = subtask.column_id === lastCol.id ? firstCol.id : lastCol.id;

  const res = await fetch(`/api/cards/${subtask.id}`, {
    method: 'PUT',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${localStorage.getItem('token')}`
    },
    body: JSON.stringify({ column_id: targetColId })
  });

  if (res.ok) {
    // Refresh subtasks
    const updated = subtasks.map(s =>
      s.id === subtask.id
        ? { ...s, column_id: targetColId, column_name: columns.find(c => c.id === targetColId)?.name || '' }
        : s
    );
    setSubtasks(updated);
    onCardUpdate?.(); // Refresh board
  }
};
```
